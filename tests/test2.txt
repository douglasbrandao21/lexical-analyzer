void sort(array vetor, int n) {
  int k, j, aux;

  for (k = 1; k < n; k++) {
    for (j = 0; j < n - 1; j++) {
      if (vetor[j] > vetor[j + 1]) {
        aux = vetor[j];
        vetor[j] = vetor[j + 1];
        vetor[j + 1] = aux;
      }
    }
  }

}

int main() {
  array vector = [1, 4, 5, 6, 9, 2];

  sort(vector);

  printf(vector);

}
